HTML básico

  • <span>: etiqueta en línea para mostrar texto o aplicar estilos sin romper el flujo.
  • data-*: atributos personalizados que puedes leer desde JS (por ejemplo, data-id).
  • id: identificador único para acceder desde JavaScript o CSS.
  • class: se usa para agrupar varios elementos con el mismo estilo o función.

Ejemplo de acceso al DOM con querySelector

Podemos acceder a elementos por etiqueta, clase o id:


// Ejemplos de acceso al DOM
document.querySelector("p")         // primera etiqueta 
    document.querySelector("#miId")     // elemento con id="miId"
    document.querySelector(".miClase")  // elemento con class="miClase"
    document.querySelectorAll("li")     // todos los

Arrays y métodos útiles


                    // Ejemplos de métodos de arrays
                    let numeros = [2, 4, 6, 8, 10, 11, 14];

                    // map() crea un nuevo array
                    let dobles = numeros.map(num => num * 2);

                    // filter() filtra según una condición
                    let pares = numeros.filter(num => num % 2 === 0);

                    // find() devuelve el primer valor que cumpla la condición
                    let encontrado = numeros.find(num => num > 8);

                    // some() comprueba si algún elemento cumple la condición
                    let hayImpares = numeros.some(num => num % 2 !== 0);

                    // every() comprueba si todos cumplen la condición
                    let todosPares = numeros.every(num => num % 2 === 0);

                    // reduce() acumula los valores (por ejemplo, suma total)
                    let suma = numeros.reduce((a, b) => a + b, 0);

                    // forEach() recorre el array
                    numeros.forEach((num, i) => console.log(i, num));

Objetos y JSON


                // Objeto tradicional
                let persona = {
                  nombre: "Ana",
                  edad: 25,
                  ciudad: "Madrid"
                };

                // Acceso a valores
                console.log(persona.nombre);

                // Objeto en formato JSON (cadena)
                let personaJSON = '{"nombre":"Ana","edad":25}';
                let personaObj = JSON.parse(personaJSON); // de texto → objeto
                let textoJSON = JSON.stringify(personaObj); // de objeto → texto

Importar y exportar (módulos JS)


// Archivo MathUtils.js
export const PI = 3.1416;
export function suma(a, b) { return a + b; }
export class Circulo {
  constructor(radio) { this.radio = radio; }
  area() { return PI * this.radio * this.radio; }
}

// Archivo principal.js
import { PI, suma, Circulo } from './MathUtils.js';
console.log(PI, suma(2, 3));
const c = new Circulo(5);
console.log(c.area());
